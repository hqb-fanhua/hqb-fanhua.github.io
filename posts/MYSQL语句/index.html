MYSQL语句

创建数据

create database [数据库] default charset utf8;

查看数据库

show database;

删除数据库

drop database [数据库];

查看表

show tables;

创建表

create table t1(
	id int auto_increment primary key,
    name varchar(50) not null
)default charset=utf8;

增加表数据

insert into t1(id,name) values('zhangsan');

insert into t1(name) values('wangwu'),('zhaoliu');

insert into t1(name) select name from t2;

清空表

delete from t1;

truncate table t1;

删除表

drop table t1;

修改表

update t1 set name="李四" where name="zhangsan";

查询表数据

select * from t1;  查询所有数据

select * from t1 where id = 1; 查询id等于1的数据

select * from t1 where id > 1 and age > 12  条件查询，id大于1并且年龄大于12

select * from t1 where id >2 or name = 'zhangsan'; 查询id大于2或者名字为'张三'的数据

select * from t1 where id in(1,2,15);  查询id是1,2,15的数据

select * from t1 where id not in(1,2,15); 查询不是1，2，15的数据

select * from t1 where id between 1 and 5; 查询id从1到5的数据数据

通配符查询

select * from t1 where name like 'zhang%'; 查询'zhang'开头的所有数据

select * from t1 where name lke '%san'; 查询后面内容包含有'san'的所有数据

限制查询

select * from t1 limit 5;  查询前5条数据

select * from t1 limit 4,5; 查询从第四行开始的5行数据

select * from t1 limit 5 offset 4; 查询从第四行开始的5行数据

排序

select * from t1 order by id desc;  从大到小，降序排序

select * from t1 order by id asc;   从小到大，升序排序

select * from t1 order by id desc limit 10; 取后10条数据

分组

select count(id),part_id from userinfo group by part_id;
---如果需要进行聚合函数的二次筛选，使用having函数
select count(id),part_id from userinfo group by part_id having  count(id) > 1;

连表查询

select * from userinfo as u left join department as d on a.part_id = d.id; 左连接查询

select * from userinfo as u right join department as d on a.part_id = d.id; 右连接查询

外键

create table uerInfo(
	uid int auto_increment primary key,
    name varchar(20) not null,
    constraint fk_user_depar foreign key("uid") references depart("id")
)engine=innodb default charset=utf8

主键

​ 数据库表主键只有一个，但是主键可以由多个列组成

create table t1(
	nid int(11) not null auto_increment,
    pid int(11) not null,
    num int(11) not null,
    primary key(nid,pid)
)engine=innodb default charset=utf8

设置自增起始值

​ 查看表详细信息

desc t1;

show create table t1 \G;

设置自增

alter table t1 auto_increment=1;

自动步长

​ 基于会话级别的

show session variables like 'auto_inc%'; 查看全局变量
set session auto_increment_increment=2;  设置会话步长

​ 基于全局级别的

show global variables like 'auto_inc%';	 查看全局变量
set session auto_increment_increment=2;  设置会话步长

唯一索引

create table t1(
	id int(11) not null auto_increment primary key,
    num int not null,
    unique uql (num)
)

唯一索引:
	约束不能重复(可以为空)
	可以加速查找

索引的变种

1.一对多关系

2.多对多的关系

导出数据文件

备份：数据表结构+数据
--mysqldump -u [用户名] [数据库名] > [输出的名字] -p
mysqldump -u root db1 > db1.sql -p

备份：数据表结构
--mysqldump -u [用户名] -d [数据库名] > db1.sql -p
mysqldump -u root -d db1 > db1.sql -p

导入数据文件

--创建数据库
create database db5;
--导入数据文件
mysqldump -u root db5 < db1.sql -p
